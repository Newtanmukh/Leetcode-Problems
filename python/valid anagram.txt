class Solution(object):
    def isAnagram(self, s, t):
        dict1={}
        dict2={}
        
        for x in s:
            if x in dict1.keys():
                dict1[x]=dict1[x]+1
            else:
                dict1[x]=1
            
        for x in t:
            if x in dict2.keys():
                dict2[x]=dict2[x]+1
            else:
                dict2[x]=1
        
        return dict1==dict2
