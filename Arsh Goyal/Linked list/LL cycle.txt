bool hasCycle(ListNode *head) {
       ListNode* fast;
         ListNode* slow;
         fast=head;
         slow=head;
       while(fast !=NULL and fast->next!=NULL)
       {
           slow=slow->next;
           fast=fast->next->next;
           if(fast==slow)
               return true;
       }
         return false;
    }
