#include <bits/stdc++.h>

#include <iostream>

using namespace std;

 int search(vector<int>& nums, int target) {
       int low=0;
       int high=nums.size()-1;
       
       while(low<=high)
       {
           int mid=low+(high-low)/2;
           
           if(nums[mid]==target)
           {return mid;}
           
           else if(nums[mid]>target)
           {low=mid+1;}
           else
           high=mid-1;
           
           
       }
              
        return -1;
    }


int main()
{
    
    //vector<int> p = {-1,0,3,5,9,12};
    vector<int> p = {45,31,29,25,24,12,9,5,3,0,-1,-4,-9,-15,-18,-23};
    int target=0;
    
    int index=search(p, target);
    
    cout<<"Index is : "<<index<<endl;

    return 0;
}
