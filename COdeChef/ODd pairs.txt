#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	
	while(t--)
	{
	    long long n;
	    cin>>n;
	    if(n%2)
	    {
	        long long  odd=(n+1)/2;
	        long long even=(n-1)/2;
	        cout<<2*odd*even<<endl;
	    }
	    else
	    {
	        long long ans=n*n/2;
	        cout<<ans<<endl;
	    }
	}
	return 0;
}
