#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	int n;
	set<int>s;
	
	while(t--)
	{
	    cin>>n;
	    vector<int>a(n,0);
	    s.clear();
	    
	    for(int i=0;i<n;i++)
	    {
	        cin>>a[i];
	        s.insert(a[i]);
	    }
	    
	    if(s.size()%2==0  )
	    cout<<"YES"<<endl;
	    else
	    {
	    if(n>s.size())
	    cout<<"YES"<<endl;
	    else
	    cout<<"NO"<<endl;
	    }
	    
	}
	return 0;
}
