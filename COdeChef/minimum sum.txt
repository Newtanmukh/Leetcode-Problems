#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	int t;
	
	cin>>t;
	while(t--)
	{
	    cin>>n;
	    vector<long long>nums(n,0);
	    
	    long long gcd;
	    for(int i=0;i<n;i++)
	    cin>>nums[i];
	    
	    gcd=__gcd(nums[0],nums[1]);
	    
	    for(int i=2;i<n;i++)
	    gcd=__gcd(gcd,nums[i]);
	    
	    cout<<n*gcd<<endl;
	    
	    
	}
	return 0;
}
