#include <bits/stdc++.h>

using namespace std;

int main() {
	int t;
	cin>>t;
	
	while(t--)
	{
	    int a;
	    int b;
	    cin>>a>>b;
	    int mini=min(a,b);
	    int maxi=max(a,b);
	    while(mini<maxi)
	    mini=mini*2;
	    
	    if(mini==maxi)
	    cout<<"Yes"<<endl;
	    else
	    cout<<"No"<<endl;
	}
	return 0;
}
