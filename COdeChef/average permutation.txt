#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	    int n;
	    cin>>n;
	    
	    if(n==3)
	    cout<<3<<" "<<2<<" "<<1<<endl;
	    else if(n==4)
	    {
	        cout<<3<<" "<<2<<" "<<1<<" "<<4<<endl;
	    }
	    else
	    {
	    vector<int>nums(n,0);
	    
	    nums[0]=n;
	    nums[n-1]=n-1;
	    
	    nums[1]=n-2;
	    nums[n-2]=n-3;
	    
	    for(int i=2;i<=n-3;i++)//check for when n-3 =2. so, n=5 onwards it would work fine. hence, write separately for n=3 and n=4.
	    {
	        nums[i]=i-1;
	    }
	    
	    for(int i=0;i<nums.size();i++)
	    {
	        cout<<nums[i]<<" ";
	    }
	    printf("\n");
	    }
	}
	return 0;
}
