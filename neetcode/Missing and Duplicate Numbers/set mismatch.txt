class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
     int  n=nums.size();
     unordered_map<int,int>mapper;
        
     vector<int>p;
        
     for(auto x:nums)
     {      
         mapper[x]++;
     }
     
     for(int i=1;i<=n;i++)
     {
         if(mapper[i]==2)
             p.push_back(i);
         if(mapper[i]==0)
             p.push_back(i);
     }
      
      return p;  
    }
};
