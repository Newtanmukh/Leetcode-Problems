class Solution {
public:
int lengthOfLongestSubstring(string s) {
    
    if(s.size()==0)
        return 0;
    
    int i=0;
    int j=0;
    unordered_map<char,int>mapper;
    int ans=INT_MIN;
    while(j<s.size())
    {
         mapper[s[j]]++;
       
         while(mapper[s[j]]>1)
         {
             mapper[s[i]]--;
             i++;             
         }
         ans=max(ans,j-i+1);
        j++;
    }
    
    return ans;
    
    
}
};
