 int count=0;
    int target;
    void traverse(TreeNode* root)
    {
        if(root==NULL)
            return;
        
        dfs(root,0);
        traverse(root->left);
        traverse(root->right);
    }
    
    void dfs(TreeNode* root,long sum)
    {
        if(!root)
            return;
        
        sum=sum+root->val;
        
        if(sum==target)
            count++;
        
        dfs(root->left,sum);
        dfs(root->right,sum);
        
    }
    
    int pathSum(TreeNode* root, int targetSum) {
        
        target = targetSum;
        traverse(root);
        return count;
    }
