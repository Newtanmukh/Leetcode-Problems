    public:
    void inorder(Node* root,int &sum)
    {
        if(!root)
        return;
        inorder(root->right,sum);
        int x=root->data;
        root->data=sum;
        sum=sum+x;
        inorder(root->left,sum);
        
        
    }
    void transformTree(struct Node *root)
    {   int x=0;
        inorder(root,x);
        
    }
