  int count=0;
    
    void dfs(TreeNode* root,int sum,int target)
    {
        if(root==NULL)
            return;
        
        if(root->left==NULL and root->right==NULL)
        {
            sum=sum+root->val;
            if(sum==target)
            count++;
            
            return;
        }
        sum=sum+root->val;
            
        dfs(root->left,sum,target);
        dfs(root->right,sum,target);
        
    }
    bool hasPathSum(TreeNode* root, int targetSum) {
        int sum=0;
        dfs(root,sum,targetSum);
        
        if(count>0)
            return true;
        else
            return false;
        
    }
