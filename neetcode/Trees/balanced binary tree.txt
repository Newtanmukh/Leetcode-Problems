class Solution {
public:
    bool ans=true;
   
    int maxheight(TreeNode* root)
    {
        if(!root)
            return 0;
        
        int left=maxheight(root->left);
        int right=maxheight(root->right);
        
        if(abs(left-right)>1)
            ans=false;
        
        return 1+max(left,right);
        
    }
    
    bool isBalanced(TreeNode* root) {
        
        int x;        
        x=maxheight(root);
        return ans;
    }
};
