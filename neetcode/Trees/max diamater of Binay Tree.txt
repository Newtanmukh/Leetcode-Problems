class Solution {
public:
    int maxi=0;
    
    int findmax(TreeNode* root)
    {
        if(!root)
            return 0;
        
        int lh=findmax(root->left);
        int rh=findmax(root->right);
        
        maxi=max(lh+rh,maxi);
            
            return 1+max(lh,rh);
        
    }
    
    int diameterOfBinaryTree(TreeNode* root) {
        int x=findmax(root);
        return maxi;
    }
};
