class Solution {
public:
    
    vector<int>p;
    
    void inorder(TreeNode* root)
    {if(!root)
        return;
     
     inorder(root->left);
     p.push_back(root->val);
     inorder(root->right);
     
        
    }
    
    int kthSmallest(TreeNode* root, int k) {
        
    inorder(root);
        return p[k-1];
        
        
    }
};
