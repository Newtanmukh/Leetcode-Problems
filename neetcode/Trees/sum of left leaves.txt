int total=0;
    void dfs(TreeNode* root,bool isleft){
        if(!root)
            return;
        
        if(isleft and root->left==NULL and root->right==NULL)
            total=total+root->val;
        
        dfs(root->left,true);
        dfs(root->right,false);
    }
    
    int sumOfLeftLeaves(TreeNode* root) {
        dfs(root,false);
        return total;
    }
