int findKthSmallest(vector<int>& nums, int k) {
        priority_queue <int> maxheap;
        
        for(auto x:nums)
        {
            maxheap.push(x);
            if(maxheap.size()>k)
            {
                maxheap.pop();
            }
            
        }
        return maxheap.top();
    }
