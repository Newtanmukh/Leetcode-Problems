int count=0;
    
    void dfs(int src,int dest,vector<int>adj[])
    {
        if(src==dest)
        {
            count++;
            return;
        }
        
        for(int u:adj[src])
        dfs(u,dest,adj);
    }
    
    int countPaths(int V, vector<int> adj[], int source, int destination) {
        dfs(source,destination,adj);
        
        return count;
    }
