	int minSwaps(vector<int>&nums)
	{
	    int size=nums.size();
	    vector<pair<int,int>> v(size);
	    int i;
	    
	    for(i=0;i<size;i++)
	    {
	        v[i]={nums[i],i};
	    }
	    sort(v.begin(),v.end());
	    int count=0;
	    
	    for(i=0;i<size;i++)
	    {
	        if(v[i].second==i)
	        continue;
	        else
	        {
	            count++;
	            swap(v[i],v[v[i].second]);
	            i--;
	        }
	        
	    }
	    return count;
	}
