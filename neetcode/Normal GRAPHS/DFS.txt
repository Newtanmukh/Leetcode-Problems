class Solution {
  public:
    // Function to return a list containing the DFS traversal of the graph.
    
    
    
    void dfs(int x,vector<bool>&visited,vector<int>&ans,vector<int> adj[])
    {
        if(visited[x]==true)
        return;
        
        visited[x]=true;
        ans.push_back(x);
        
        for(auto y:adj[x])
        {
            if(!visited[y])
            dfs(y,visited,ans,adj);
        }
        
    }
    
    vector<int> dfsOfGraph(int V, vector<int> adj[]) {
        
        vector<bool>visited(V+1,false);
        vector<int>ans;
        
        for(int i=0;i<V;i++)
        {
            
            dfs(i,visited,ans,adj);
        }
        
        return ans;
    }
};
