  bool ispalindrome(string s,int x,int y)
    {
        while(x<y)
        {
            if(s[x]==s[y])
            {
                x++;
                y--;
            }
            else
            {
                return false;
            }
        }
        return true;
        
    }
    
    bool validPalindrome(string s) {
      
        
        int i=0;
        int j=s.size()-1;
        
        while(i<j)
        {
            if(s[i]==s[j])
            {
                i++;
                j--;
            }
            else
            {
                return ispalindrome(s,i+1,j)||ispalindrome(s,i,j-1);
            }
        }
        return true;
    }
