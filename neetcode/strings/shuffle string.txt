class Solution {
public:
    
    static bool sortbyindex(const pair<int,char> &a, const pair<int,char>&b)
    {
        return a.first<b.first?true:false;
    }    
    
    string restoreString(string s, vector<int>& indices) {
        
        vector<pair<int,char>> vec;
        
        string t="";

        for(int i=0;i<indices.size();i++)
        {
            vec.push_back(make_pair(indices[i],s[i]));
        }
        
        sort(vec.begin(),vec.end(),sortbyindex);
        
        for(auto x:vec)
        {
            t=t+x.second;
        }
        
        
        
        return t;
        
    }
};
