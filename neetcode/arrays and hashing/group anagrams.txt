class Solution {
public:
    
    
    
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
    
        map<string,vector<string>>mapper;
        
        vector<vector<string>>ans;
        
        for(auto &x:strs)
        {string temp=x;
         sort(temp.begin(),temp.end());
         
         mapper[temp].push_back(x);   
        }
        
        for(auto &x:mapper)
        {
            ans.push_back(x.second);
        }
        return ans;
        
        
    }
};
