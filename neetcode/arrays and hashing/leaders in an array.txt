  vector<int> leaders(int a[], int n){
        vector<int>ans;
        if(n==1)
        {
            ans.push_back(a[0]);
            return {a[0]};
        }
        
        int maxi=INT_MIN;
        
        
        
        for(int i=n-1;i>=0;i--)
        {
            if(a[i]>=maxi)
            {ans.push_back(a[i]);
                maxi=a[i];
            }
            
        }
        
        reverse(ans.begin(),ans.end());
        return ans;
    }
