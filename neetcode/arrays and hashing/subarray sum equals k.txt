int subarraySum(vector<int>& nums, int k) {
    
    int sum=0;
    unordered_map<int,int>mapper;
    
    int count=0;
    int size=nums.size();
    
    for(int i=0;i<size;i++)
    {
        sum=sum+nums[i];
        
        if(sum==k)
            count++;
        
        if(mapper.count(sum-k))
            count=count+mapper[sum-k];
        
        mapper[sum]++;
    }
    return count;
        
    }
