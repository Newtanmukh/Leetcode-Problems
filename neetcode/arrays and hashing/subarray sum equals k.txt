int subarraySum(vector<int>& nums, int k) {
       unordered_map<int,int>mp;
        int count=0;
    int sum=0;
    
    for(auto x:nums)
    {
        sum=sum+x;
        
        if(sum==k)
        {
            count=count+1;
        }
        
        if(mp.count(sum-k)==1)
        {
            count=count+mp[sum-k];
        }
        
        mp[sum]++;
        
    }
    
    return count;
        
    }
