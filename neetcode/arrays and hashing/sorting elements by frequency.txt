#include <bits/stdc++.h>
using namespace std;

bool comp(pair<int,int>a,pair<int,int>b)
{
   if(a.second == b.second)return a.first < b.first;
  return a.second>b.second;
   
}
int main() {

int t,n;
cin>>t;
while(t--)
{
    cin>>n;
    vector<int>v(n);
    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }
    map<int,int>mp;
    for(int i=0;i<n;i++)
    mp[v[i]]++;
    
    vector<pair<int,int>>ans;
    for(auto it:v)
    {
        ans.push_back({it,mp[it]});
    }
    sort(ans.begin(),ans.end(),comp); 
    
    for(auto it:ans)
    {
        cout<<it.first<<" ";
    }
   
    cout<<endl;
    
}
return 0;
}
