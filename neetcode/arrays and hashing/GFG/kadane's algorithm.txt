long long max(long long x,long long y)
    {
        if(x>y)
        return x;
        return y;
    }
    
    long long maxSubarraySum(int arr[], int n){
        
        vector<long long>dp(n,0);
        
        dp[0]=arr[0];
        
        for(int i=1;i<n;i++)
        {
            dp[i]=max(dp[i-1]+arr[i],arr[i]);
        }
        return *max_element(dp.begin(),dp.end());
    }
