 int firstRepeated(int arr[], int n) {
        map<int,int>m;
        int index=-1;
        
        for(int i=0;i<n;i++)
        {
            m[arr[i]]++;
            
        }
        
        for(int i=0;i<n;i++)
        {
            if(m[arr[i]]>1)
            return i+1;
        }
        return index;
    }
