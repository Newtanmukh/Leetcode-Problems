int findPlatform(int arr[], int dep[], int n)
    {
    	sort(arr,arr+n);
    	sort(dep,dep+n);
    	int i=1;
    	int j=0;
    	int ans=0;
    	int platform=1;
    	
    	while(i<n and j<n)
    	{
    	    if(arr[i]<=dep[j])
    	    {
    	        platform++;
    	        i++;
    	    }
    	    else
    	    {
    	        platform--;
    	        j++;
    	    }
    	    ans=max(ans,platform);
    	}
    	return ans;
    }
