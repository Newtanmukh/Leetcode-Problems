long long subCount(long long arr[], int N, long long k)
	{
	    unordered_map<int,int>mp;
	    int sum=0;
	    int count=0;
	    
	    mp[0]++;
	    
	    for(int i=0;i<N;i++)
	    {
	        sum=sum+arr[i];
	        mp[(sum%k+k)%k]++;
	    }
	    
	    for(auto x:mp)
	    {
	        count=count+x.second*(x.second-1)/2;
	    }
	    return count;
	    
	}
