 int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
        
        map<int,int> mapper;
        
        for(auto x:nums1){
            for(auto y:nums2){
                mapper[x+y]++;
            }
        }
        
        
        int count=0;
        for(auto x:nums3){
            for(auto y:nums4){
               count+=mapper[-x-y];
            }
        }
        
        return count;
        
        
    }
