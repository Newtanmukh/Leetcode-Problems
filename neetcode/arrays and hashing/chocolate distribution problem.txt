long long min(long long x,long long y)
    {
        if(x>y)
        return y;
        return x;
    }
    
    long long findMinDiff(vector<long long> a, long long n, long long m)
    {
    
    long long ans;
    sort(a.begin(),a.end());
    
    
    ans=a[m-1]-a[0];
    
    for(int i=m;i<n;i++)
    {
        ans=min(ans,a[i]-a[i-m+1]);
    }
    
    return ans;
    
    }
