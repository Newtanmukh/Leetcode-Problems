 int countQuadruplets(vector<int>& nums) {
        int cnt = 0;
        int n = nums.size();
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                for(int k=j+1;k<n;k++){
                    for(int f=k+1;f<n;f++){
                        if(nums[i]+nums[j]+nums[k]==nums[f]){
                            cnt++;
                        }
                    }
                }
            }
        }
        return cnt;
    }
