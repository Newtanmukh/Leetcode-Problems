 bool isPalindrome(ListNode* head) {
        
      vector<int>ans;
      while(head)
      {
          ans.push_back(head->val);
          head=head->next;
      }
        
      for(int i=0;i<ans.size()/2;i++)
      {
          if(ans[i]!=ans[ans.size()-1-i])
              return false;
      }
        
        return true;
    }
