  ListNode* removeElements(ListNode* head, int target) {
     
      if(!head)
          return head;
        
      ListNode* node=head;
        
      while(node!=NULL and node->next!=NULL)
      {
          if(node->next->val==target)
              node->next=node->next->next;
          else
              node=node->next;
      }
        
        if(head->val==target)
        {
            head=head->next;
        }
        return head;
    }
