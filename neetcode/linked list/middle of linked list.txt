 ListNode* middleNode(ListNode* head) {
       ListNode *slow = head, *fast = head;//this will take care of the edge case of the head=NULL thing as well.
        while (fast && fast->next)
            slow = slow->next, fast = fast->next->next;
        return slow;
    }















class Solution {
public:
    ListNode* middleNode(ListNode* head) {
       if(head==NULL)
           return NULL;
        
        ListNode* node=head;
        int middle;
        middle=0;
        
        while(node)
        {
            middle++;
            node=node->next;
            
        }
        
        middle=middle/2;
        
        while(middle--)
        {
            head=head->next;
        }
        
        return head;
        
    }
};
