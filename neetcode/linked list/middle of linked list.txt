class Solution {
public:
    ListNode* middleNode(ListNode* head) {
       if(head==NULL)
           return NULL;
        
        ListNode* node=head;
        int middle;
        middle=0;
        
        while(node)
        {
            middle++;
            node=node->next;
            
        }
        
        middle=middle/2;
        
        while(middle--)
        {
            head=head->next;
        }
        
        return head;
        
    }
};
