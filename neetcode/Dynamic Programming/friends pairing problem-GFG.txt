int countFriendsPairings(int n) 
    { 
        if(n==1)
        return 1;
        
        int mod=pow(10,9)+7;
        
        if(n==2)
        return 2;
        
        long dp[n+1];
        dp[0]=0,dp[1]=1,dp[2]=2;
        
        for(int i=3;i<=n;i++)
         dp[i]=(dp[i-2]*(i-1)+dp[i-1])%1000000007;
        
        return dp[n];
    }
