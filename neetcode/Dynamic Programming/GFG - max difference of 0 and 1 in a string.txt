	int maxSubstring(string s)
	{
	  vector<int>nums(s.size(),0);
	  
	  for(int i=0;i<s.size();i++)
	  {
	      if(s[i]=='0')
	      nums[i]=1;
	      else
	      nums[i]=-1;
	  }
	  
	  int dp[s.size()]={0};
	  
	  dp[0]=nums[0];
	  
	  for(int i=1;i<s.size();i++)
	  {
	      dp[i]=max(nums[i],dp[i-1]+nums[i]);
	  }
	  
	  return *max_element(dp,dp+s.size());
	   
	}
