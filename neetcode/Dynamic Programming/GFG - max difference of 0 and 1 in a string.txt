int maxSubstring(string s)
	{
	   vector<int>nums(s.size(),0);
	   
	   for(int i=0;i<s.size();i++)
	   {
	       if(s[i]=='0')
	       nums[i]=1;
	       else
	       nums[i]=-1;
	   }
	   
	   vector<int>dp(s.size(),0);
	   
	   dp[0]=nums[0];
	   
	   for(int i=1;i<nums.size();i++)
	   {
	       dp[i]=max(dp[i-1]+nums[i],nums[i]);
	   }
	   
	   return *max_element(dp.begin(),dp.end());
	   
	}
