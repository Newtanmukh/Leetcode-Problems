long long countWays(int n, int k){
        // code here
        
        if(n==1)
        return k;
        
        if(n==2)
        return k*k;
        
        long long dp[n+1];
        dp[0]=0;
        dp[1]=k;
        dp[2]=k*k;
        
        for(int i=3;i<n+1;i++)
        {
            dp[i]=(k-1)*(dp[i-1]+dp[i-2])%1000000007;
        }
        
        return dp[n];
    }
