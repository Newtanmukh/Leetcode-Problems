 vector<int> minPartition(int V)
    {
        vector<int>v={1,2,5,10,20,50,100,200,500,2000};
      vector<int>vv;
      int n=10;
      for(int i=n-1;i>=0;i--)
      {
        while(V>=v[i])
        {
            V-=v[i];
            vv.push_back(v[i]);
        }
        
      }
      return vv;
    }
