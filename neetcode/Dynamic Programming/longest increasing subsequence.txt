int lengthOfLIS(vector<int>& nums) {
   int lis[nums.size()]; 

	lis[0] = 1; 


	for (int i = 1; i < nums.size(); i++ ) 
	{ 
		lis[i] = 1; 
		for (int j = 0; j < i; j++ ) 
			if ( nums[i] > nums[j] ) 
				lis[i] = max(lis[i],lis[j] + 1); //lis[i] denotes the maximum subsequence upto i including index i.
	} 

	return *max_element(lis, lis+nums.size()); 
            }
