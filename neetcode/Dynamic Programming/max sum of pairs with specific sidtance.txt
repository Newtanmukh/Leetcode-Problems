int maxSumPairWithDifferenceLessThanK(int arr[], int n, int k)
   {
       int sum=0;
       sort(arr,arr+n);
       for(int i=n-1;i>=1;)
       {
           if(arr[i]-arr[i-1]<k)
           {
               sum=arr[i]+arr[i-1]+sum;
               i=i-2;
           }
           else
           i=i-1;
           
       }
       return sum;
   }
