 cpp_int findCatalan(int n) 
    {
        vector<cpp_int>catalan(n+1,0);
        
        if(n==1||n==0)
        return 1;
        
        
        catalan[0]=1;
        catalan[1]=1;
        //catalan[2]=2;
        
        for(int i=2;i<=n;i++)
        {
            for(int j=0;j<i;j++)
            {
               catalan[i]=catalan[i]+catalan[i-j-1]*catalan[j]; 
            }
        }
        return catalan[n];
        
        
    }
