class Solution {
public:
    int numTrees(int n) {
        if(n<=1)
            return 1;
 
	unsigned long int catalan[n + 1];


	catalan[0] = catalan[1] = 1;

	
	for (int i = 2; i <= n; i++) {
		catalan[i] = 0;
		for (int j = 1; j < i+1; j++)
			catalan[i] += catalan[j-1] * catalan[i - j];
	}


	return catalan[n];
    }
};
