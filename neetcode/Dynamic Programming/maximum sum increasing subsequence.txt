int maxSumIS(int a[], int n)  
	{  
	   
	   vector<int>ans;
	   
	   for(int i=0;i<n;i++)
	   {
	       ans.push_back(a[i]);
	   }
	   
	   for(int i=1;i<n;i++)
	    for(int j=0;j<i;j++)
	    {
	        if(a[i]>a[j])
	        ans[i]=max(ans[i],ans[j]+a[i]);
	    }
	    
	    return *max_element(ans.begin(),ans.end());
	   
	}
